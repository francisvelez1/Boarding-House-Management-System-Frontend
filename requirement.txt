Create file .env

Paste this:
# URL of your FastAPI backend
# Your backend main.py runs on port 8080
VITE_API_URL=http://localhost:*****/api

# Frontend base URL (used only if you need it in code)
VITE_FRONTEND_URL=http://localhost:*****

# App metadata (optional but handy)
VITE_APP_NAME="Boarding House Management System"
VITE_APP_ENV=development


npm install
npm install vue-router
npm install pinia

npm run dev
